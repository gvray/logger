📖 快速预览

import { Logger, LogLevel } from '@gvray/logger';

const logger = new Logger({
  level: LogLevel.DEBUG,
  timestamp: 'time',
  colors: true
});

// 基础日志
logger.info('Server started on port', 3000);
logger.warn('API rate limit:', { remaining: 10, limit: 100 });
logger.error('Connection failed:', new Error('timeout'));

// 命名空间
const db = logger.child('database');
db.info('Connected to MongoDB');

// 中间件
logger.use(prefixMiddleware('[API]'));
logger.info('Request received');
